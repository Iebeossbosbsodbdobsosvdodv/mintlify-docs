### JavaScript

```javascript
async function getAccountInfo() {
  const response = await fetch(
    'https://api.zynxwallet.com.br/v1/account/info',
    {
      headers: {
        'X-API-Key': process.env.ZYNX_API_KEY
      }
    }
  );

  const data = await response.json();
  
  if (!data.success) {
    throw new Error(data.message);
  }

  return data.data;
}

// Usar

console.log(`Nome: ${account.name}`);
console.log(`Email: ${account.email}`);
console.log(`Plano: ${account.plan}`);
```

### Python

```python
import requests

def get_account_info():
    response = requests.get(
        'https://api.zynxwallet.com.br/v1/account/info',
        headers={
            'X-API-Key': os.getenv('ZYNX_API_KEY')
        }
    )

    data = response.json()
    
    if not data['success']:
        raise Exception(data['message'])

    return data['data']

# Usar
account = get_account_info()
print(f"Nome: {account['name']}")
print(f"Email: {account['email']}")
print(f"Plano: {account['plan']}")
```
